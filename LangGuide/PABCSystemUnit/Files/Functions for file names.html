Functions for file names

function ExpandFileName(fname: string): string;
        Returns the full name of the file fname
function ExtractFileDir(fname: string): string;
        Extracts the directory substring from the full file name
function ExtractFileDrive(fname: string): string;
        Extracts the drive substring from the full file name
function ExtractFileExt(fname: string): string;
        Extracts the extension substring from the full file name
function ExtractFileName(fname: string): string;
        Extracts the file name substring from the full file name
function ExtractFilePath(fname: string): string;
        Extracts the path substring from the full file name
